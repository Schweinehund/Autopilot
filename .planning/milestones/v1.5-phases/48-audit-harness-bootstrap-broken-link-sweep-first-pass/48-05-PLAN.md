function readFile(relPath) {
  const abs = join(process.cwd(), relPath);
  if (!existsSync(abs)) return null;
  return readFileSync(abs, 'utf8').replace(/\r\n/g, '\n');
}

        const stderr = err.stderr ? err.stderr.toString() : '';
        const isMissing = err.code === 'ENOENT' || err.status === 127

        if (isMissing) return { pass: true, skipped: true, detail: 'node not found -- skipped' };
        return { pass: false, detail: '--self-test FAIL: ' + stderr.slice(0, 200) };

**Step 5 — Add runner loop (clone from check-phase-30.mjs lines 303-337; verbatim with name change):**

```javascript
let failed = 0;
let skipped = 0;
function padLabel(s) { return s + ' '.repeat(Math.max(0, 60 - s.length)); }
console.log('check-phase-48 -- Phase 48 deliverables\n');
for (const check of checks) {
  const prefix = '[' + check.id + '/' + checks.length + '] ' + check.name + ': ';
  const result = check.run();
  const showDetail = result.detail && (VERBOSE || !result.pass || result.skipped);
  if (result.skipped) {
    skipped++;
    process.stdout.write(padLabel(prefix) + 'SKIPPED' + (showDetail ? ' ' + result.detail : '') + '\n');
  } else if (result.pass) {
    process.stdout.write(padLabel(prefix) + 'PASS' + (showDetail ? ' ' + result.detail : '') + '\n');
  } else {
    failed++;
    process.stdout.write(padLabel(prefix) + 'FAIL' + (showDetail ? ' ' + result.detail : '') + '\n');
  }
}
console.log('\nResult: ' + (checks.length - failed - skipped) + ' PASS, ' + failed + ' FAIL, ' + skipped + ' SKIPPED');
process.exit(failed > 0 ? 1 : 0);
```
