**PowerShell recipe (verbatim from 66-RESEARCH.md Code Examples Pattern 3 + 66-PATTERNS.md "Auditor-Independence via Fresh-Context Spawn" section):**

```powershell
# 1. Generate 8-char random temp-dir suffix (66-RESEARCH.md Don't Hand-Roll table — Get-Random built-in)
$rand = -join ((48..57) + (97..122) | Get-Random -Count 8 | ForEach-Object {[char]$_})
$auditPath = "$env:TEMP\v1.6-audit-$rand"

# 2. Capture HEAD SHA from main repo BEFORE clone (audit metadata)
$mainHeadSha = git -C D:\claude\Autopilot rev-parse HEAD

# 3. Fresh clone (--no-hardlinks for stricter physical isolation -- separate .git/ blobs)
git clone --no-hardlinks D:\claude\Autopilot $auditPath

# 4. Capture clone HEAD SHA to confirm it matches main repo HEAD
$cloneHeadSha = git -C $auditPath rev-parse HEAD
if ($cloneHeadSha -ne $mainHeadSha) {
    Write-Error "Clone HEAD SHA ($cloneHeadSha) does not match main repo HEAD SHA ($mainHeadSha) -- abort"
    exit 1
}

# 5. Move into clone
Push-Location $auditPath

# 6. Run each validator, capture exit codes + last summary line
$results = [ordered]@{}
$cmds = @(
    @{ name = 'harness';        path = 'scripts/validation/v1.6-milestone-audit.mjs' },
    @{ name = 'harness-selftest'; path = 'scripts/validation/v1.6-milestone-audit.mjs'; args = @('--self-test') },
    @{ name = 'check-phase-62'; path = 'scripts/validation/check-phase-62.mjs' },
    @{ name = 'check-phase-63'; path = 'scripts/validation/check-phase-63.mjs' },
    @{ name = 'check-phase-64'; path = 'scripts/validation/check-phase-64.mjs' },
    @{ name = 'check-phase-65'; path = 'scripts/validation/check-phase-65.mjs' },
    @{ name = 'check-phase-66'; path = 'scripts/validation/check-phase-66.mjs' }
)
foreach ($c in $cmds) {
    $argList = @($c.path)
    if ($c.ContainsKey('args')) { $argList += $c.args }
    $stdoutCombined = & node @argList 2>&1
    $exit = $LASTEXITCODE
    $stdoutText = ($stdoutCombined | Out-String)
    $summary = ($stdoutCombined | Select-String -Pattern 'Summary:|Result:' | Select-Object -Last 1).ToString()
    if (-not $summary) { $summary = ($stdoutCombined | Select-Object -Last 1 | Out-String).Trim() }
    $results[$c.name] = @{
        exit = $exit
        summary = $summary
        stdout_tail = ($stdoutCombined | Select-Object -Last 20 | Out-String).Trim()
    }
    Write-Host "[$($c.name)] exit=$exit | $summary"
}

# 7. Persist results JSON for Wave 5 consumption
$resultsJsonPath = "$env:TEMP\v1.6-audit-results-$rand.json"
$results | ConvertTo-Json -Depth 4 | Out-File $resultsJsonPath -Encoding utf8

# 8. Return to main repo BEFORE cleanup
Pop-Location

# 9. Cleanup throwaway clone (no orphan temp directories)
Remove-Item -Recurse -Force $auditPath

# 10. Report final state
Write-Host "`n=== AUDIT COMPLETE ==="
Write-Host "HEAD SHA: $mainHeadSha"
Write-Host "Results JSON: $resultsJsonPath"
Write-Host "Clone removed: $auditPath"
```
