    ```
    cat > /tmp/androidDocPaths-probe.mjs <<'EOF'
    // Re-declare androidDocPaths logic without importing the harness (file-reads-only; avoid side effects).
    // Mirror the post-Plan-02 function; this is a sanity-spot-check, not the actual harness.
    import { readdirSync, statSync, existsSync } from 'node:fs';
    import { join } from 'node:path';
    import process from 'node:process';

    function walkMd(dir) {
      const abs = join(process.cwd(), dir);
      if (!existsSync(abs)) return [];
      const results = [];
      (function walk(current) {
        let entries;
        try { entries = readdirSync(current); } catch { return; }
        for (const entry of entries) {
          const full = join(current, entry);
          let stat; try { stat = statSync(full); } catch { continue; }
          if (stat.isDirectory()) walk(full);
          else if (entry.endsWith('.md')) results.push(full.replace(/\\/g, '/'));
        }
      })(abs);
      return results;
    }
    function hasUnderscoreDirSegment(relPath) {
      const segs = relPath.split('/');
      return segs.slice(0, -1).some(s => /^_/.test(s));
    }
    // Enumerate the Android scope the same way v1.4.1 harness does.
    const enumerated = [
      ...walkMd('docs/android-lifecycle'),
      ...walkMd('docs/admin-setup-android'),
      'docs/_glossary-android.md',
      'docs/_templates/admin-template-android.md',
    ];
    const filtered = enumerated.filter(p => !hasUnderscoreDirSegment(p));
    const relFiltered = filtered.map(p => p.replace(process.cwd().replace(/\\/g,'/') + '/', ''));
    const has_glossary = relFiltered.some(p => /\/_glossary-android\.md$/.test(p) || /^_glossary-android\.md$/.test(p) || /docs\/_glossary-android\.md$/.test(p));
    const has_template = relFiltered.some(p => /_templates\/admin-template-android\.md$/.test(p));
    console.log('glossary included:', has_glossary);
    console.log('template included:', has_template);
    process.exit((has_glossary && !has_template) ? 0 : 1);
    EOF
    node /tmp/androidDocPaths-probe.mjs
    RC=$?
    rm /tmp/androidDocPaths-probe.mjs
    test $RC -eq 0 && echo "SCOPE-FILTER OK" || echo "SCOPE-FILTER FAIL — investigate"
    ```
