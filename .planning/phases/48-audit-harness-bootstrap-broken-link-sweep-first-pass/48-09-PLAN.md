  const { execFileSync } = require('node:child_process');

**Step 3 — Parse the sweep output:**

```
FILE: docs/admin-setup-android/04-cobo-knox-me.md

[✖] #non-existent-anchor → Status: 400
[✓] https://github.com/example → Status: 200
```

Extract `[✖]` (broken) entries and the FILE: line they belong to. Use a Node script to convert sweep-output.txt to a structured findings list:

```bash
node -e "
  const fs = require('node:fs');
  const lines = fs.readFileSync('/tmp/phase-48/sweep-output.txt','utf8').split('\n');
  let currentFile = '';
  const findings = [];
  for (const line of lines) {
    const fileMatch = line.match(/^FILE:\s*(.+)$/);
    if (fileMatch) { currentFile = fileMatch[1].trim(); continue; }
    const brokenMatch = line.match(/^\\[✖\\]\s*(\S+)\s*→\s*Status:\s*(\d+)/);
    if (brokenMatch && currentFile) {
      findings.push({ file: currentFile, target: brokenMatch[1], status: brokenMatch[2] });
    }
  }
  fs.writeFileSync('/tmp/phase-48/findings.json', JSON.stringify(findings, null, 2));
  console.log('Total findings:', findings.length);
  // Categorize
  const catA = findings.filter(f => f.target.startsWith('#') || /\.md#/.test(f.target));
  const catB = findings.filter(f => /\.md$/.test(f.target) || /\.\.\//.test(f.target));
  const catC_candidates = []; // populated by manual review
  console.log('Category A (anchors) candidates:', catA.length);
  console.log('Category B (file paths) candidates:', catB.length);
"
```
